XPLOT_TITLE

      Put a title above all PDC/DTF subplots with measure and alpha-value.

Contents

  • Syntax
  • Input Arguments
  • Output

Syntax

XPLOT_TITLE(alpha,metric,measure)

Input Arguments

alpha:      significance level

measure:    should be either 'pdc'| 'dtf'

metric:     'euc'  -- Euclidean   ==> |PDC|^2  or |DTF|^2
            'diag' -- diagonal    ==> |gPDC|^2 or |DC|^2
            'info' -- information ==> |iPDC|^2 or |iDTF|^2

Output

Title on above all subplots of DTF/PDC.

See also XPLOT, PVALUES_XPLOT xplot | xplot.html |

% (C) Koichi Sameshima & Luiz A. Baccala, 2021. See file license.txt in
% installation directory for licensing terms.
%

function [] = xplot_title(alpha,metric,measure)

if nargin == 1,
    c=alpha;
    if isfield(c,'dtf2')
        measure = 'DTF';
        alpha = c.alpha;
        metric =  c.metric;
    elseif isfield(c,'pdc2')
        measure = 'PDC';
        alpha = c.alpha;
        metric =  c.metric;
    end
elseif nargin < 3,
    measure = 'PDC';
end;

if exist('c') ~=0
   alphapercentage = 100*(c.alpha);
else
   alphapercentage = 100*alpha;
end;

alphastr = int2str(alphapercentage);
measure = upper(measure);
metric = lower(metric);
switch metric
    case 'euc'
        %nop
    case 'diag'
        if measure == 'DTF',
            measure = 'DC';    % generalized DTF or Directed Coherence (DC)
        else
            measure = '_gPDC'; % generalized PDC
        end;
    case 'info'
        measure = ['_i' measure];
    otherwise
        error('Unknown metric.')
end;

if alpha <= 0,
   suptitle(['{|' measure '|}^2'])
elseif alpha < 0.01,
   suptitle(['{|' measure '|}^2 (' '{\alpha = ' sprintf('%g',alpha) '})'])
else
   suptitle(['{|' measure '|}^2 (' '{\alpha = ' alphastr '%}' ')'])
end;
